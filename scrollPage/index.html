xScrollPage
方法 说明 参数 示例
render(selector,onfun) 根据DOM节点自动加载 @className: 类名 @onfun 回调函数 源码
示例源码
<script type="text/javascript">
	
 xUtils.addEvent(window, 'load', function() 
 {
 	//如果有jquery会启用jquery方式，没有则原生JS滚动模式
   xScrollPage.render('#test',function()
   {
   	alert('加载更多数据')
   	  for (var i = 0; i < 20 ;i++) {
   		 jQuery('<p>'+i+'</p>').insertBefore(jQuery('#test'))
   	}
   	   
   });
 })
</script>
