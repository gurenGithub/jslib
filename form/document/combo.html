autocompete-ajax 加载
autocompete-loadAreaData 自定义数据 
<div class="xForm">
                        <div class="control">
                            <span class="prefix">autocompete-loadAreaData 自定义数据</span>
                            <span class="autocomplete"
                                  tfield='text'
                                  vfield='value'
                                  loaddata='loadAreaData'>
                                <input type="text" />
                            </span>
                        </div>
                        <script type="text/javascript">
                            function loadAreaData(text) {
                                var me = this;
                                var data = [{ text: "天河区", value: 1, city: 1 }, { text: "南宁区", value: "2", city: 2 }];
                                setTimeout(function () {
                                    var _newData = [];
                                    for (var i = 0; i < data.length; i++) {
                                        var item = data[i];
                                        if (item.text.indexOf(text) > -1) {
                                            _newData.push(item);
                                        }
                                    }
                                    me.refresh(_newData);
                                }, 2000
                                )
                            }
                        </script>
                    </div>
