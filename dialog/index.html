title:'标题' ,

width : '宽度(400)',

height : '高度(300)',

buttons:'按钮([{title,click}])',

content:'内容',

src:'http路径',

onClose:'关闭事件'

}

title:'测试',

height:500,

alert('弹出层关闭');

}});
